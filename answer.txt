#include <iostream>
#include <vector>
using namespace std;
long long n;
long long k;
/**/
long long check(vector<long long> answ, long long ans){
    long long l = answ[0];
    long long c = 1;
    for(long long i = 0; i < n; i ++){
        if(answ[i] >= l + ans){
            c += 1;
            l = answ[i];
        }
    }
    return c;
}

long long maximum(vector<long long> a){
    long long maxim   = a[0];
    for (long long i = 0; i < n; i++){
        if(a[i] > maxim ){
            maxim = a[i];
        }
    }
    return maxim;
}

int main(){
    long long m, x;
    cin >> n >> k;

    vector<long long> stoila(n);
    for(long long i = 0; i < n; i++){

        cin >>stoila[i];

    }
    //n = n - 1;


    long long l = 0, r = maximum(stoila);
    while (r - l > 1){

        m = (r + l) / 2;
        if (check(stoila, m) >=  k){
            l = m;
        }
        else{
            r = m;
        }
    }
    cout << l;
    return 0;
}
